public class StudentController {

    // Current student
    public Student__c stu {get; set;}

    // List of students
    public List<Student__c> stuList {get; set;}

    // Selected Id
    public Id selectedId {get; set;}

    // Constructor
    public StudentController() {

        stu = new Student__c();

        loadStudents();
    }

    // Load all students
    public void loadStudents() {

        stuList = [
            SELECT Id,
                   Name,
                   Roll_No__c,
                   Class__c,
                   Mobile_No__c
            FROM Student__c
            ORDER BY CreatedDate DESC
        ];
    }

    // INSERT
    public void insertStudent() {

        insert stu;

        stu = new Student__c();

        loadStudents();
    }

    // EDIT
    public void editStudent() {

        stu = [
            SELECT Id,
                   Name,
                   Roll_No__c,
                   Class__c,
                   Mobile_No__c
            FROM Student__c
            WHERE Id = :selectedId
        ];
    }

    // UPDATE
    public void updateStudent() {

        update stu;

        stu = new Student__c();

        loadStudents();
    }

    // DELETE  (simple and same as product logic)
    public void deleteStudent() {

        Student__c s = [
            SELECT Id
            FROM Student__c
            WHERE Id = :selectedId
        ];

        delete s;

        loadStudents();
    }
}
